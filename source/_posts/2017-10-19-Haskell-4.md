# Maximum

```haskell
max :: (Ord a) => [a] -> a
max [] = error "max of empty list"
max [x] = x
max (x:xs)
  | x > maxTail = x
  | otherwise = maxTail
  where maxTail = max xs
```

```haskell
deplicate :: (Num i, Ord i) => i -> a -> [a]
deplicate n x
  | n <= 0 = []
  | otherwise = x:replicate (n - 1) x

zip' :: [a] -> [b] -> [(a,b)]  
zip' _ [] = []  
zip' [] _ = []  
zip' (x:xs) (y:ys) = (x,y):zip' xs ys
```
